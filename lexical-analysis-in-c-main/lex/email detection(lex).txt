int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <inputfile> <outputfile>\n", argv[0]);
        exit(1);
    }

    yyin = fopen(argv[1], "r");
    if (!yyin) {
        perror("Error opening input file");
        exit(1);
    }

    out = fopen(argv[2], "w");
    if (!out) {
        perror("Error opening output file");
        exit(1);
    }

    yylex();   /* Run lexical analysis */

    fclose(yyin);
    fclose(out);
    return 0;
}
